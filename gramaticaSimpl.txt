S' -> PROGRAM
PROGRAM -> HEADER main ( ) { DECLLIST LISTA_INSTRUCTIUNI }
HEADER -> #include<iostream>
DECLLIST -> DECL ; | DECL ; DECLLIST
DECL -> TYPE id = const	| TYPE id
TYPE -> int
LISTA_INSTRUCTIUNI -> INSTRUCTIUNE | INSTRUCTIUNE LISTA_INSTRUCTIUNI
INSTRUCTIUNE -> ATRIBUIRE ;
ATRIBUIRE -> id = const
ATRIBUIRE -> id = id
