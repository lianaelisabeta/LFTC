S' -> PROGRAM
PROGRAM -> HEADER int main ( ) { BODY }
HEADER -> #include<iostream> #include<string> | #include<iostream> | #include<string>
BODY -> DECLLIST LISTA_INSTRUCTIUNI | DECLLIST
DECLLIST -> DECL ; | DECL ; DECLLIST
DECL -> TYPE id = const	| TYPE id [ const ] | TYPE id
TYPE -> int | char | double | std::string
LISTA_INSTRUCTIUNI -> INSTRUCTIUNE | INSTRUCTIUNE LISTA_INSTRUCTIUNI
INSTRUCTIUNE -> INSTRUCTIUNE_SIMPLA
INSTRUCTIUNE_SIMPLA -> ATRIBUIRE ;
ATRIBUIRE -> id = EXPRESIE
EXPRESIE -> id | const
